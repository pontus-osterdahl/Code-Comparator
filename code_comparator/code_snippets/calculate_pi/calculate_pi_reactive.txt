int threadCount = 4;

int n = 100000;

Observable.range(0, threadCount)
  .flatMap(i -> Observable.just(i)
    .subscribeOn(Schedulers.computation())
    .map(i2 -> calcPi(threadCount, i2, n)))
  .scan((x, y) -> x + y)
  .lastElement()
  .subscribe(e -> System.out.println(e * 4));

  try
  {
    Thread.sleep(1000);
  }

  catch(InterruptedException e)
  {

  }
